def run
  say_status 'success', "Showing unison logs from your sync container: #{@unison.get_container_name}", :green
  cmd = "docker exec #{@unison.get_container_name} tail -F /tmp/unison.log"
  @watch_thread = thread_exec(cmd, 'Sync Log:')
end
